71   // Load the question and options
72   function loadQuestion() {
73       document.getElementById("question").textContent = questions[currentQuestion].question;
74       document.getElementById("option1").textContent = questions[currentQuestion].options[0];
75       document.getElementById("option2").textContent = questions[currentQuestion].options[1];
76       document.getElementById("option3").textContent = questions[currentQuestion].options[2];
77       document.getElementById("option4").textContent = questions[currentQuestion].options[3];
78   
79       // Play the appropriate background music
80       playBackgroundMusic();
81   
82       // Update money tree highlighting
83       updateMoneyTreeHighlighting();
84   }

86   // Update money tree highlighting based on the current question
87   function updateMoneyTreeHighlighting() {
88       let moneyTreeItems = document.querySelectorAll('#money-tree-list li');
89   
90       // Remove all existing highlights
91       moneyTreeItems.forEach((li) => li.classList.remove('highlighted'));
92   
93       // Apply highlighting based on the current question
94       if (currentQuestion >= 0 && currentQuestion < 4) {
95           moneyTreeItems[3].classList.add('highlighted'); // Highlight "Pick the next player"
96       } else if (currentQuestion >= 4 && currentQuestion < 7) {
97           moneyTreeItems[2].classList.add('highlighted'); // Highlight "Extra 10 minutes break"
98       } else if (currentQuestion >= 7 && currentQuestion < 9) {
99           moneyTreeItems[1].classList.add('highlighted'); // Highlight "Free Snack"
100      } else if (currentQuestion === 9) { // Highlight NO HOMEWORK for the last question
101          moneyTreeItems[0].classList.add('highlighted'); // Highlight "NO HOMEWORK"
102      }
103  }

105  // Play background music based on the current question
106  function playBackgroundMusic() {
107      stopAllTracks(); // Stop any currently playing tracks
108  
109      if (currentQuestion < 4) {
110          audioTracks.track1.play(); // Play track1 for the first 4 questions
111      } else if (currentQuestion < 6) {
112          audioTracks.track2.play(); // Play track2 for the next 2 questions
113      }
114  }

116  // Stop all audio tracks
117  function stopAllTracks() {
118      Object.values(audioTracks).forEach(track => {
119          track.pause();
120          track.currentTime = 0;
121      });
122  }

124  // Check the answer
125  function checkAnswer(option) {
126      if (askTheClassTimer) {
127          clearInterval(askTheClassTimer);
128          askTheClassTimer = null;
129          stopAllTracks(); // Stop timer track if answering
130      }
131  
132      if (option === questions[currentQuestion].correct) {
133          score++;
134          currentQuestion++;
135          if (currentQuestion < questions.length) {
136              loadQuestion();
137          } else {
138              stopAllTracks();
139              let prize = moneyTree[moneyTree.length - 1];
140              if (prize === "NO HOMEWORK") {
141                  audioTracks.track4.play(); // Play winning track
142              }
143              alert("Congratulations! You won " + prize + "!");
144              resetGame();
145          }
146      } else {
147          stopAllTracks();
148          alert("Sorry, wrong answer! You won " + moneyTree[currentQuestion > 0 ? currentQuestion - 1 : 0] + "!");
149          resetGame();
150      }
151  }

153  // Lifeline: Phone a Friend
154  function phoneAFriend() {
155      if (lifelines.phoneAFriend) {
156          stopAllTracks(); // Stop any playing background music

160      } else {

162      }
163  }

165  // Lifeline: Ask the Class (with a 30s countdown timer)
166  function askTheClass() {
167      if (lifelines.askTheClass) {
168          lifelines.askTheClass = false;
169          document.getElementById("ask").disabled = true;
170  
171          stopAllTracks(); // Stop any playing background music
172          audioTracks.track3.play(); // Play timer track
173  
174          alert("You have 30 seconds to ask the class, you then must answer.Press ok to start the timer");
175  
176          // Display the countdown timer
177          document.getElementById("timer").textContent = `Time left: ${timeLeft} seconds`;
178          askTheClassTimer = setInterval(function() {
179              timeLeft--;
180              document.getElementById("timer").textContent = `Time left: ${timeLeft} seconds`;
181  
182              if (timeLeft <= 0) {
183                  clearInterval(askTheClassTimer);
184                  askTheClassTimer = null;
185                  audioTracks.track3.pause(); // Stop timer track
186                  audioTracks.track3.currentTime = 0;
187                  alert("Time's up! You didn't pick an answer in time.");
188                  resetGame();
189              }
190          }, 1000); // 1000ms = 1 second
191      } else {
192          alert("You already used Ask the Class!");
193      }
194  }

196  // Lifeline: 50/50
197  function fiftyFifty() {
198      if (lifelines.fiftyFifty) {
199          let correctAnswer = questions[currentQuestion].correct;
200          let optionsToDisable = [];
201  
202          // Find two incorrect options to disable (hide)
203          for (let i = 0; i < 4; i++) {
204              if (i !== correctAnswer && optionsToDisable.length < 2) {
205                  optionsToDisable.push(i);
206              }
207          }
208  
209          // Hide the two incorrect options
210          optionsToDisable.forEach(optionIndex => {
211              document.getElementById("option" + (optionIndex + 1)).style.visibility = "hidden";
212          });
213  
214          lifelines.fiftyFifty = false;
215          document.getElementById("fifty").disabled = true;
216      } else {
217          alert("You already used 50/50!");
218      }
219  }

221  // Reset the game
222  function resetGame() {
223      currentQuestion = 0;
224      score = 0;
225      timeLeft = 30;
226      lifelines = { phoneAFriend: true, askTheClass: true, fiftyFifty: true };
227  
228      // Re-enable all options and lifelines
229      for (let i = 1; i <= 4; i++) {
230          document.getElementById("option" + i).disabled = false;
231          document.getElementById("option" + i).style.visibility = "visible"; // Reset visibility
232      }
233      document.getElementById("phone").disabled = false;
234      document.getElementById("ask").disabled = false;
235      document.getElementById("fifty").disabled = false;
236  
237      resetMoneyTreeHighlighting(); // Ensure money tree resets
238      loadQuestion(); // Load the first question
239  }
